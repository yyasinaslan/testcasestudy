    --header "Content-Type: application/json" \

    "http://localhost/api/discounts"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

    method: "GET",
    headers,

Indirim kurali ekle

Example request:
curl --request POST \
    "http://localhost/api/discounts" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"name\": \"vel\",
    \"type\": \"cart\",
    \"filters\": \"{\\\"category\\\":2}\",
    \"rule\": \"expedita\",
    \"priority\": 10
}"

const url = new URL(
    "http://localhost/api/discounts"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "name": "vel",
    "type": "cart",
    "filters": "{\"category\":2}",
    "rule": "expedita",
    "priority": 10
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());

Request      

POST api/discounts

Body Parameters

name  string optional  

type  string optional  

product | cart.

filters  json optional  

Optional. You can filter by category or product id.

rule  json  

rule.condition  string  

rule.amount  string  

priority  integer optional  

Priority. Higher priority discounts will be calculated first.
