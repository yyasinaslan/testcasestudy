Example request:
curl --request POST \
    "http://localhost/api/orders" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"consumerId\": 14,
    \"items\": [
        {
            \"productId\": 11,
            \"quantity\": 3,
            \"unitPrice\": 1765.9477595,
            \"total\": 0.846325
        }
    ],
    \"total\": 0
}"

const url = new URL(
    "http://localhost/api/orders"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "consumerId": 14,
    "items": [
        {
            "productId": 11,
            "quantity": 3,
            "unitPrice": 1765.9477595,
            "total": 0.846325
        }
    ],
    "total": 0
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
