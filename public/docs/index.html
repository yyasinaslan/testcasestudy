Siparis Ekleme

Example request:
curl --request POST \
    "http://localhost/api/orders" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"consumerId\": 16,
    \"items\": [
        {
            \"productId\": 6,
            \"quantity\": 18,
            \"unitPrice\": 11626617.165713,
            \"total\": 240.45582458
        }
    ],
    \"total\": 28873163.40565
}"

const url = new URL(
    "http://localhost/api/orders"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "consumerId": 16,
    "items": [
        {
            "productId": 6,
            "quantity": 18,
            "unitPrice": 11626617.165713,
            "total": 240.45582458
        }
    ],
    "total": 28873163.40565
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
